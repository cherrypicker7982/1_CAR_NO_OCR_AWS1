3.도커 업데이트.
# 1) 기존 컨테이너 정리(떠있으면)

# 2) 이미지 다시 빌드 (requirements 안 바꿨으면 --no-cache 생략 가능)
docker build -t lp-ocr .
docker build --no-cache -t lp-ocr .

# 3) 컨테이너 재실행
docker run -d --name lp-ocr --restart=always -p 80:8000 lp-ocr
